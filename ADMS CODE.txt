CREATE TABLE DONOR
       (ID NUMBER(3) CONSTRAINT PK_DONOR PRIMARY KEY,
	DNAME VARCHAR2(14) ,
	AGE NUMBER(2),
        ADDRESS VARCHAR2(14),
        SEX VARCHAR2(8),
        PHONE_NO NUMBER(11),
        TYPE_OF_DONOR VARCHAR(10),
        DONATION_DATE DATE) ;

CREATE TABLE BLOOD
       (CODE NUMBER(3) CONSTRAINT PK_BLOOD PRIMARY KEY,
	BAG_NO NUMBER(3) ,
	BLOOD_GROUP VARCHAR2(3),
        COST NUMBER(4) ) ;

CREATE TABLE BLOOD_BANK_MANAGER
       (ID NUMBER(3) CONSTRAINT PK_BLOOD_BANK_MANAGER PRIMARY KEY,
	MNAME VARCHAR2(14) ,
	PHONE_NO NUMBER(11),
        ADDRESS VARCHAR(20) ) ;




CREATE TABLE EMPLOYEE
       (ID NUMBER(3) CONSTRAINT PK_EMPLOYEE PRIMARY KEY,
	ENAME VARCHAR2(14) ,
	E_SAL NUMBER(11),
        E_POS VARCHAR(10),
        ADDRESS VARCHAR2(20) ) ;




CREATE TABLE EQUIPMENT
       (EQ_ID NUMBER(3) CONSTRAINT PK_EQUIPMENT PRIMARY KEY,
	EQTYPE VARCHAR2(14) ,
	EQ_QUANT NUMBER(3),
        COST NUMBER(10) ) ;


CREATE TABLE BLOOD_BANK
       (ORDER NUMBER(3) CONSTRAINT PK_BLOOD_BANK PRIMARY KEY,
	BLOOD_GROUP VARCHAR2(3),
	BAG_NO NUMBER(3),
        ISSUE VARCHAR2(3) );
